import numpy as np
import statsmodels.formula.api as smf

# 过滤汽车类数据
video_noncar = video[video['类别'] != '汽车']

# 定义发布时间分段函数
def get_period(hour):
    if 20 <= hour or hour < 6:
        return '夜间'
    elif 6 <= hour < 9:
        return '早间'
    elif 9 <= hour < 12:
        return '上午'
    elif 12 <= hour < 14:
        return '中午'
    elif 14 <= hour < 17:
        return '下午'
    else:
        return '傍晚'
video_noncar['视频发布时间'] = video_noncar['发布时间'].apply(get_period)

# 对数转换
video_noncar['log_点赞数'] = np.log(video_noncar['点赞数'])
video_noncar['log_分享数'] = np.log(video_noncar['分享数'])
video_noncar['log_评论数'] = np.log(video_noncar['评论数'])

# 线性回归模型
formula = '''
log_点赞数 ~ C(视频发布时间, Treatment("夜间")) + log_分享数 + log_评论数 + 
时长 + C(类别, Treatment("宠物")) + 标题字数
'''
model = smf.ols(formula, data=video_noncar).fit()
print(model.summary())
